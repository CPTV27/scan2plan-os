import {
  HoverCard,
  HoverCardContent,
  HoverCardTrigger,
} from "@/components/ui/hover-card";
import { HelpCircle } from "lucide-react";

interface ContextHelpProps {
  content: string;
}

export function ContextHelp({ content }: ContextHelpProps) {
  return (
    <HoverCard>
      <HoverCardTrigger asChild>
        <HelpCircle className="h-4 w-4 text-slate-400 hover:text-blue-600 cursor-help inline-block ml-2 align-middle" />
      </HoverCardTrigger>
      <HoverCardContent className="w-80 text-sm text-slate-600 bg-white border-slate-200 shadow-lg">
        {content}
      </HoverCardContent>
    </HoverCard>
  );
}
